## Alert Dialog 警告对话框

import { AlertDialog } from "solid-element-ui";

<AlertDialog
    title="确认删除？"
    description="此操作不可逆。"
    trigger={<button className="text-red-600">点击弹出AlertDialog</button>}
/>

### 导入

```tsx
import { AlertDialog } from "solid-element-ui";
```

### 基础用法

```tsx
<AlertDialog
    title="确认删除？"
    description="此操作不可逆。"
    trigger={<button className="text-red-600">删除</button>}
    action={
        <button className="bg-red-600 text-white px-4 py-2 rounded">
            确定
        </button>
    }
/>
```
